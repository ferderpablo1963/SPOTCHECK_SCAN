function processImageOCR(event) { const file = event.target.files[0]; if (!file) return; const ocrIcon = document.getElementById('btn-ocr-icon'); const ocrText = document.getElementById('btn-ocr-text'); // 1. Activamos el estado de carga visual ocrIcon.className = "fa-solid fa-spinner fa-spin text-xs text-amber-400"; ocrText.innerText = "LEYENDO..."; const reader = new FileReader(); reader.onload = function(e) { const img = new Image(); img.onload = function() { // 2. Ejecutamos Tesseract con manejo seguro de promesas Tesseract.recognize( img, 'eng', // Puedes cambiar a 'spa' si tus reportes usan palabras en español { logger: m => console.log("Progreso OCR:", m) // Te permite ver en la consola si avanza o dónde se traba } ) .then(({ data: { text } }) => { let textoBruto = text.toLowerCase(); console.log("--- TEXTO BRUTO DEL SISTEMA ---", textoBruto); // Normalización de distorsiones de caracteres de la cámara let textoLimpio = textoBruto .replace(/tb1/g, 'tbl') .replace(/tbi/g, 'tbl') .replace(/tb\|/g, 'tbl') .replace(/[\/\\]/g, ' '); let mesasEncontradas = []; let regexMesa = /tbl\s*(\d{1,3})/g; let coincidencia; while ((coincidencia = regexMesa.exec(textoLimpio)) !== null) { let numExtraido = parseInt(coincidencia[1], 10).toString(); if (!mesasEncontradas.includes(numExtraido)) { mesasEncontradas.push(numExtraido); } } console.log("Números de mesas identificados:", mesasEncontradas); let totalPintadas = 0; tables.forEach(t => { let identificadorMesa = t.label.toString().trim(); if (mesasEncontradas.includes(identificadorMesa)) { t.status = 'occupied'; totalPintadas++; } }); showToast(totalPintadas > 0 ? `Se activaron ${totalPintadas} mesas en el plano` : "No se detectaron nuevas mesas en la foto"); render(); saveData(); }) .catch(err => { // 3. Si Tesseract falla, atrapamos el error para que no cuelgue la App console.error("Error crítico en el reconocimiento OCR:", err); showToast("Error al procesar la imagen. Intente de nuevo."); }) .finally(() => { // 4. PASE LO QUE PASE, devolvemos el botón a su estado original (Destrabe) ocrIcon.className = "fa-solid fa-camera text-xs"; ocrText.innerText = "SCAN"; // Limpiamos el input para permitir volver a subir la misma foto si se desea document.getElementById('ocr-camera-input').value = ""; }); }; img.src = e.target.result; }; reader.readAsDataURL(file); }
